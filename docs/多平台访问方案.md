```typescript
// frontend/src/service-worker.ts
/// <reference lib="webworker" />

import { precacheAndRoute } from 'workbox-precaching'
import { registerRoute } from 'workbox-routing'
import { CacheFirst, StaleWhileRevalidate } from 'workbox-strategies'
import { ExpirationPlugin } from 'workbox-expiration'

// 缓存核心资源
precacheAndRoute(self.__WB_MANIFEST)

// 缓存静态资源
registerRoute(
  ({ request, url }) => {
    return request.destination === 'style' ||
           request.destination === 'manifest' ||
           request.destination === 'script' ||
           request.destination === 'worker'
  },
  new StaleWhileRevalidate({
    cacheName: 'static-resources',
  })
)

// 缓存 API 请求
registerRoute(
  ({ url }) => url.pathname.startsWith('/api/'),
  new CacheFirst({
    cacheName: 'api-cache',
    plugins: [
      new ExpirationPlugin({
        maxEntries: 100,
        maxAgeSeconds: 7 * 24 * 60 * 60, // 7 天
      }),
    ],
  })
)

// 缓存图片
registerRoute(
  ({ request }) => request.destination === 'image',
  new CacheFirst({
    cacheName: 'image-cache',
    plugins: [
      new ExpirationPlugin({
        maxEntries: 50,
        maxAgeSeconds: 30 * 24 * 60 * 60, // 30 天
      }),
    ],
  })
)
```

export async function subscribeToPush() {
  const registration = await navigator.serviceWorker.ready

  const subscription = await registration.pushManager.subscribe({
    userVisibleOnly: true,
    applicationServerKey: urlBase64ToUint8Array(
      'YOUR_VAPID_PUBLIC_KEY'

  })

  // 将 subscription 发送到后端保存
  await saveSubscription(subscription)

  return subscription
}

#### 3.2.3 触摸手势支持

```typescript
// frontend/src/composables/useSwipe.ts
import { ref, onMounted, onUnmounted } from 'vue'

export function useSwipe(element: Ref<HTMLElement | null>) {
  const startX = ref(0)
  const startY = ref(0)
  const isSwiping = ref(false)

  const onTouchStart = (e: TouchEvent) => {
    startX.value = e.touches[0].clientX
    startY.value = e.touches[0].clientY
    isSwiping.value = true
  }

  const onTouchMove = (e: TouchEvent) => {
    if (!isSwiping.value) return

    const diffX = e.touches[0].clientX - startX.value
    const diffY = e.touches[0].clientY - startY.value

    // 判断滑动方向
    if (Math.abs(diffX) > Math.abs(diffY)) {
      // 水平滑动
      if (diffX > 50) {
        emit('swipe-right')
      } else if (diffX < -50) {
        emit('swipe-left')
      }
    }
  }

  const onTouchEnd = () => {
    isSwiping.value = false
  }

  onMounted(() => {
    if (element.value) {
      element.value.addEventListener('touchstart', onTouchStart)
      element.value.addEventListener('touchmove', onTouchMove)
      element.value.addEventListener('touchend', onTouchEnd)
    }
  })

  onUnmounted(() => {
    if (element.value) {
      element.value.removeEventListener('touchstart', onTouchStart)
      element.value.removeEventListener('touchmove', onTouchMove)
      element.value.removeEventListener('touchend', onTouchEnd)
    }
  })

  return { isSwiping }
}

import { app, BrowserWindow, Tray, Menu } from 'electron'
import path from 'path'

let mainWindow: BrowserWindow | null = null

function createWindow() {
  mainWindow = new BrowserWindow({
    width: 1200,
    height: 800,
    webPreferences: {
      nodeIntegration: true,
      contextIsolation: false,
    },
    icon: path.join(__dirname, '../frontend/public/icons/icon.png'),
  })

  if (process.env.NODE_ENV === 'development') {
    mainWindow.loadURL('http://localhost:3000')
    mainWindow.webContents.openDevTools()
  } else {
    mainWindow.loadFile(path.join(__dirname, '../frontend/dist/index.html'))
  }

  mainWindow.on('closed', () => {
    mainWindow = null
  })
}

function createTray() {
  tray = new Tray(path.join(__dirname, '../frontend/public/icons/tray.png'))

  const contextMenu = Menu.buildFromTemplate([
    { label: '打开应用', click: () => mainWindow?.show() },
    { label: '退出', click: () => app.quit() },
  ])

  tray.setContextMenu(contextMenu)
  tray.setToolTip('PubHAgent')
}

app.whenReady().then(() => {
  createWindow()
  createTray()
})

# backend/api/routes/api_keys.py
from fastapi import APIRouter, Depends, HTTPException
from datetime import datetime, timedelta
import secrets

router = APIRouter(prefix="/api/keys", tags=["API Keys"])

@router.post("/create")
async def create_api_key(user: User = Depends(get_current_user)):
    """创建 API Key"""
    api_key = f"pk_{secrets.token_urlsafe(32)}"

    # 保存到数据库
    db_api_key = APIKey(
        key=api_key,
        user_id=user.id,
        expires_at=datetime.now() + timedelta(days=365),
    )
    db.add(db_api_key)
    db.commit()

    return {
        "api_key": api_key,
        "expires_at": db_api_key.expires_at,
    }

@router.get("/list")
async def list_api_keys(user: User = Depends(get_current_user)):
    """列出所有 API Keys"""
    keys = db.query(APIKey).filter(APIKey.user_id == user.id).all()
    return [
        {
            "id": key.id,
            "key_prefix": key.key[:10] + "...",
            "created_at": key.created_at,
            "expires_at": key.expires_at,
            "last_used": key.last_used,
        }
        for key in keys
    ]

@router.delete("/revoke/{key_id}")
async def revoke_api_key(
    key_id: str,
    user: User = Depends(get_current_user),
):
    """撤销 API Key"""
    key = db.query(APIKey).filter(
        APIKey.id == key_id,
        APIKey.user_id == user.id,
    ).first()

    if not key:
        raise HTTPException(404, "API Key 不存在")

    key.revoked = True
    db.commit()

    return {"message": "API Key 已撤销"}
```

### 4.2 Nginx 配置

```nginx
# nginx.conf
server {
    listen 80;
    server_name your-domain.com;

    # 重定向到 HTTPS
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name your-domain.com;

    ssl_certificate /etc/nginx/ssl/fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/privkey.pem;

    # 前端静态文件
    location / {
        root /app/frontend/dist;
        try_files $uri $uri/ /index.html;

        # PWA 支持
        add_header Cache-Control "public, max-age=31536000";
    }

    # API 代理
    location /api {
        proxy_pass http://web:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # WebSocket 代理
    location /ws {
        proxy_pass http://web:8000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;

        # WebSocket 超时设置
        proxy_read_timeout 86400;
    }

    # 静态资源
    location /static {
        proxy_pass http://web:8000;
        add_header Cache-Control "public, max-age=86400";
    }
}
```

## 七、总结

### 7.1 强烈推荐方案（优先级排序）

1. **PWA 渐进式 Web 应用** ⭐⭐⭐⭐⭐
   - 开发成本低，收益高
   - 支持离线、推送等原生功能
   - 无需应用商店审核

2. **响应式 Web 设计** ⭐⭐⭐⭐⭐
   - 基础优化，必须实施
   - 提升移动端体验
   - 维护成本极低

3. **API 开放平台** ⭐⭐⭐⭐
   - 扩展应用场景
   - 提高项目价值
   - 支持第三方集成

4. **社交媒体分享** ⭐⭐⭐⭐
   - 易于传播
   - 降低使用门槛
   - 适合展示分析结果

### 7.2 可选方案

- **微信小程序**: 如果目标用户主要在微信生态
- **React Native/Flutter**: 如果需要原生体验和设备访问
- **Electron**: 如果需要桌面端离线使用

### 7.3 实施建议

1. **先实施 PWA + 响应式设计**（2-3 周）
   - 快速提升移动端体验
   - 开发成本低，见效快

2. **再实施 API 开放平台**（1 周）
   - 支持第三方集成
   - 扩展应用场景

3. **根据用户需求决定后续方案**
   - 如果微信用户多 → 开发小程序
   - 如果需要原生体验 → 开发 App
   - 如果桌面端需求大 → 开发 Electron
